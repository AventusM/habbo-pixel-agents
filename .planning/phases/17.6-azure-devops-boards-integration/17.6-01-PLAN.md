import type { KanbanCard } from './agentTypes.js';

/** Map Azure DevOps work item state to kanban status */
export function mapAzureDevOpsState(state: string): string {
  // Lookup table: Azure DevOps state → KanbanCard status
  const TODO = ['New', 'Proposed', 'Approved'];
  const IN_PROGRESS = ['Active', 'Committed', 'In Progress'];
  const DONE_STATES = ['Resolved', 'Done', 'Closed'];
  if (TODO.includes(state)) return 'Todo';
  if (IN_PROGRESS.includes(state)) return 'In Progress';
  if (DONE_STATES.includes(state)) return 'Done';
  if (state === 'Removed') return 'Removed';
  return 'No Status';
}

Implementation details:
- Guard: if any of org/project/pat is empty, return []
- PAT auth header: `Basic ${Buffer.from(':' + pat).toString('base64')}`
- Step 1: POST WIQL query to `https://dev.azure.com/{org}/{project}/_apis/wit/wiql?api-version=7.1`
  - WIQL: `SELECT [System.Id] FROM WorkItems WHERE [System.TeamProject] = '{project}' AND [System.State] <> 'Closed' AND [System.State] <> 'Removed' ORDER BY [System.ChangedDate] DESC`
  - Parse response: extract `workItems[].id` array
  - If empty, return []
- Step 2: POST to `https://dev.azure.com/{org}/_apis/wit/workitemsbatch?api-version=7.1`
  - Body: `{ ids: workItemIds.slice(0, 100), fields: ['System.Id', 'System.Title', 'System.State'] }`
  - Parse response: map `value[]` to KanbanCard using mapAzureDevOpsState
  - Filter out items where status === 'Removed'
- Wrap entire function body in try/catch, catch returns []
- Log errors: `console.error('[azureDevOpsBoards] fetchAzureDevOpsCards failed:', err)`

Note on WIQL filter: The WIQL WHERE clause excludes Closed and Removed items. However, the mapAzureDevOpsState function should still handle these states in case the WIQL filter is changed later. The batch response filter for 'Removed' is a safety net.
