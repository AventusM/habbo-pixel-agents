Create `src/isoNameTagRenderer.ts` with the following exports:

1. **NameTagSpec interface:**
   - name: string (agent/terminal name to display)
   - status: 'idle' | 'active' | 'waiting' | 'error' (maps to dot color)
   - anchorX: number (avatar head screen X position)
   - anchorY: number (avatar head screen Y position - top of avatar sprite)

2. **drawNameTag function:**
   - Map status to colors:
     - idle: '#00ff00' (green)
     - active: '#ffff00' (yellow)
     - waiting: '#888888' (grey)
     - error: '#ff0000' (red)
   - Set ctx.font = '8px "Press Start 2P"' before measureText
   - Measure name text width: ctx.measureText(name).width
   - Calculate pill dimensions: width = dotRadius*2 + padding + textWidth + padding, height = 16px
   - Position pill centered above anchor: pillX = anchorX - width/2, pillY = anchorY - 24px (above speech bubble position)
   - Draw pill background: ctx.fillStyle = 'rgba(0, 0, 0, 0.7)', ctx.roundRect(pillX, pillY, width, height, height/2 for pill shape), ctx.fill()
   - Draw status dot: ctx.fillStyle = statusColor, ctx.arc(pillX + padding + dotRadius, pillY + height/2, dotRadius, 0, Math.PI*2), ctx.fill()
   - Draw name text: ctx.fillStyle = (status === 'active' ? '#ffff00' : '#ffffff'), ctx.textBaseline = 'middle', ctx.fillText(name, Math.floor(pillX + dotRadius*2 + padding*1.5), Math.floor(pillY + height/2))

Constants: dotRadius=3, padding=6, pillHeight=16

// Render name tags (above avatars, before speech bubbles)
for (const avatar of renderState.current.avatars) {
  const { x: screenX, y: screenY } = tileToScreen(avatar.tileX, avatar.tileY, avatar.tileZ);
  const headY = screenY - AVATAR_HEIGHT;

  // Map avatar state to name tag status
  const status = avatar.state === 'idle' ? 'idle' : 'active'; // walk/spawning = active

  ctx.save();
  ctx.translate(renderState.current.cameraOrigin.x, renderState.current.cameraOrigin.y);
  drawNameTag(ctx, {
    name: avatar.id,
    status,
    anchorX: screenX,
    anchorY: headY,
  });
  ctx.restore();
}
