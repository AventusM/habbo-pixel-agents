<feature>
  <name>Heightmap parser, HSB colour utilities, and depth sort</name>
  <files>src/isoTypes.ts, tests/isoTypes.test.ts</files>
  <behavior>
    parseHeightmap(str):
      - '0' -> { height: 0 } (walkable, ground level)
      - '9' -> { height: 9 } (walkable, max height)
      - 'x' -> null (void)
      - 'X' -> null (void)
      - unknown char -> null (treat as void)
      - rows split on \n or \r\n
      - width = max row length (uneven rows: short rows padded with null)
      - output: TileGrid { tiles: Array<Array<{height:number}|null>>, width, height }

    hsbToHsl({ h, s, b }) where h:0-360, s:0-100, b:0-100:
      - b=100, s=0 -> l=100, sHsl=0   (white, no saturation)
      - b=100, s=100 -> l=50, sHsl=100  (full saturation)
      - b=50, s=50 -> l=37, sHsl=~50  (mid-tone)

    tileColors(hsb) -> { top: string, left: string, right: string }:
      - top face: base lightness
      - left face: base lightness - 10 (clamped to 0)
      - right face: base lightness - 20 (clamped to 0)
      - format: "hsl(H, S%, L%)"

    depthSort(renderables[]) -> renderables sorted ascending by tileX+tileY+tileZ*0.001:
      - [ {1,0,0}, {0,1,0} ] -> same sort key 1.0, stable (same-key order preserved)
      - [ {2,2,0}, {1,1,9} ] -> 4.009 vs 2.009 -> {1,1,9} first
      - [ {0,0,5}, {1,0,0} ] -> 0.005 vs 1.0 -> {0,0,5} first (stair tile stays behind)
  </behavior>
