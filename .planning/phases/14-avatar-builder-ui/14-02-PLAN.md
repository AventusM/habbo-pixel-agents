**2. Modify `src/extension.ts`:**

Add two message handlers in the `panel.webview.onDidReceiveMessage` switch:

```typescript
case 'saveAvatar': {
  const { agentId, outfit } = msg as any;
  const wsDir = workspaceDir;
  if (wsDir) {
    const avatarsDir = join(wsDir, '.habbo-agents');
    const avatarsPath = join(avatarsDir, 'avatars.json');
    // Read existing data
    let data: Record<string, any> = {};
    try {
      if (existsSync(avatarsPath)) {
        data = JSON.parse(readFileSync(avatarsPath, 'utf8'));
      }
    } catch { /* fresh file */ }
    // Merge
    if (!data[agentId]) data[agentId] = {};
    data[agentId].outfit = outfit;
    // Write
    if (!existsSync(avatarsDir)) {
      const { mkdirSync } = require('fs');
      mkdirSync(avatarsDir, { recursive: true });
    }
    writeFileSync(avatarsPath, JSON.stringify(data, null, 2));
  }
  break;
}
case 'loadAvatars': {
  const wsDir = workspaceDir;
  if (wsDir) {
    const avatarsPath = join(wsDir, '.habbo-agents', 'avatars.json');
    let data: Record<string, any> = {};
    try {
      if (existsSync(avatarsPath)) {
        data = JSON.parse(readFileSync(avatarsPath, 'utf8'));
      }
    } catch { /* no file yet */ }
    panel.webview.postMessage({ type: 'avatarOutfits', outfits: data } as ExtensionMessage);
  }
  break;
}
```

Note: Use `fs.mkdirSync` (already imported at top of file) instead of require. The `fs` imports `readFileSync, writeFileSync, existsSync` are already present. Add `mkdirSync` to the import if not already there.
