**Implementation (from 03-RESEARCH.md Pattern 1):**

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as zlib from 'zlib';

export function extractNitro(nitroPath: string, outDir: string): void {
  // Read entire .nitro file into buffer
  const buffer = fs.readFileSync(nitroPath);
  let offset = 0;

  // Read file count (BigEndian UI16)
  const fileCount = buffer.readUInt16BE(offset);
  offset += 2;

  console.log(`Extracting ${fileCount} files from ${nitroPath}...`);

  for (let i = 0; i < fileCount; i++) {
    // Read filename length (BigEndian UI16)
    const nameLen = buffer.readUInt16BE(offset);
    offset += 2;

    // Read filename string (UTF-8)
    const filename = buffer.toString('utf8', offset, offset + nameLen);
    offset += nameLen;

    // Read compressed data length (BigEndian UI32)
    const compressedLen = buffer.readUInt32BE(offset);
    offset += 4;

    // Extract compressed data slice
    const compressedData = buffer.subarray(offset, offset + compressedLen);
    offset += compressedLen;

    // Inflate (decompress) using zlib
    const decompressed = zlib.inflateSync(compressedData);

    // Write to output directory
    const outPath = path.join(outDir, filename);
    fs.mkdirSync(outDir, { recursive: true });
    fs.writeFileSync(outPath, decompressed);

    console.log(`  ✓ ${filename} (${decompressed.length} bytes)`);
  }

  console.log(`Extraction complete: ${fileCount} files written to ${outDir}`);
}
```

**CLI wrapper for npm script invocation:**

```typescript
// If executed directly (not imported), run CLI mode
if (import.meta.url === `file://${process.argv[1]}`) {
  const nitroPath = process.argv[2];
  const outDir = process.argv[3] || 'assets/extracted';

  if (!nitroPath) {
    console.error('Usage: node extractNitro.ts <path-to.nitro> [output-dir]');
    process.exit(1);
  }

  if (!fs.existsSync(nitroPath)) {
    console.error(`Error: File not found: ${nitroPath}`);
    process.exit(1);
  }

  try {
    extractNitro(nitroPath, outDir);
  } catch (error) {
    console.error('Extraction failed:', error);
    process.exit(1);
  }
}
```

**CRITICAL: Use BigEndian methods**

- `readUInt16BE()` NOT `readUInt16LE()`
- `readUInt32BE()` NOT `readUInt32LE()`

Per Pitfall 1 from research: .nitro uses BigEndian byte order. Using LittleEndian methods causes garbage values.

**Error handling:**

- Let zlib.inflateSync() throw on invalid compressed data (corrupted .nitro)
- Let fs.readFileSync() throw if .nitro file not found
- Caller (prebuild script) catches errors and aborts build

import * as fs from 'fs';
import * as zlib from 'zlib';
import * as path from 'path';
import { extractNitro } from '../src/scripts/extractNitro.js';

describe('extractNitro', () => {
  let tempDir: string;

  beforeEach(() => {
    tempDir = fs.mkdtempSync(path.join(process.cwd(), 'tmp-nitro-test-'));
  });

  afterEach(() => {
    fs.rmSync(tempDir, { recursive: true, force: true });
  });

  it('parses single-file .nitro correctly', () => {
    // Create minimal PNG data (PNG magic bytes + minimal IHDR)
    const pngData = Buffer.from([
      0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A, // PNG signature
      // Minimal IHDR chunk would go here (skipped for test simplicity)
    ]);

    const compressed = zlib.deflateSync(pngData);
    const filename = 'test.png';

    // Build .nitro buffer manually
    const buffer = Buffer.alloc(2 + 2 + filename.length + 4 + compressed.length);
    let offset = 0;

    buffer.writeUInt16BE(1, offset); // FileCount
    offset += 2;

    buffer.writeUInt16BE(filename.length, offset); // NameLen
    offset += 2;

    buffer.write(filename, offset, 'utf8'); // Name
    offset += filename.length;

    buffer.writeUInt32BE(compressed.length, offset); // CompressedLen
    offset += 4;

    compressed.copy(buffer, offset); // ZlibData

    // Write buffer to temp .nitro file
    const nitroPath = path.join(tempDir, 'test.nitro');
    fs.writeFileSync(nitroPath, buffer);

    // Extract
    const outDir = path.join(tempDir, 'out');
    extractNitro(nitroPath, outDir);

    // Verify
    const extractedPath = path.join(outDir, filename);
    expect(fs.existsSync(extractedPath)).toBe(true);
    const extractedData = fs.readFileSync(extractedPath);
    expect(extractedData).toEqual(pngData);
  });

  // ... more tests
});

- Second argument: `assets/spritesheets` (output directory)
