interface AudioContext {
  state: 'suspended' | 'running' | 'closed';
  resume(): Promise<void>;
  decodeAudioData(arrayBuffer: ArrayBuffer): Promise<AudioBuffer>;
  createBufferSource(): AudioBufferSourceNode;
  destination: AudioDestinationNode;
}

interface AudioBuffer {
  duration: number;
  numberOfChannels: number;
  sampleRate: number;
}

interface AudioBufferSourceNode {
  buffer: AudioBuffer | null;
  connect(destination: AudioDestinationNode): void;
  start(when?: number): void;
}

**init() method (AUDIO-03 - autoplay policy compliance):**
- Return early if already initialized
- Create AudioContext if null
- Check if context.state === 'suspended', call await context.resume()
- Set initialized = true
- Log 'AudioContext initialized, state: [state]' to console

**loadSound(uri: string) method (AUDIO-02, AUDIO-04 - graceful loading):**
- Return null if context not initialized (silent fallback)
- Check cache.has(uri), return cached AudioBuffer if exists
- Wrap in try/catch block:
  - fetch(uri), check response.ok, get arrayBuffer
  - Validate arrayBuffer.byteLength > 0
  - await context.decodeAudioData(arrayBuffer)
  - cache.set(uri, audioBuffer)
  - Log success with duration
  - Return audioBuffer
- Catch block: console.warn with message 'Audio loading failed for [uri], continuing silently:', error
- Return null (silent fallback per AUDIO-04)

**play(audioBuffer: AudioBuffer | null) method:**
- Return early if audioBuffer is null OR context is null OR context.state !== 'running'
- Create source = context.createBufferSource()
- Set source.buffer = audioBuffer
- source.connect(context.destination)
- source.start(0)

**Export:** `export class AudioManager { ... }`
